 648:      */
 649:     protected function parseTokensInEntityFile($src)
 650:     {
 651:         $tokens = token_get_all($src);
 652:         $lastSeenNamespace = "";
 653:         $lastSeenClass = false;
 654: 
 655:         $inNamespace = false;
 656:         $inClass = false;
 657: 
 658:         for ($i = 0; $i < count($tokens); $i++) {
 659:             $token = $tokens[$i];
 660:             if (in_array($token[0], array(T_WHITESPACE, T_COMMENT, T_DOC_COMMENT))) {
 661:                 continue;
 662:             }
 663: 
 664:             if ($inNamespace) {
 665:                 if ($token[0] == T_NS_SEPARATOR || $token[0] == T_STRING) {
 666:                     $lastSeenNamespace .= $token[1];
 667:                 } elseif (is_string($token) && in_array($token, array(';', '{'))) {
 668:                     $inNamespace = false;
 669:                 }
 670:             }
 671: 
 672:             if ($inClass) {
 673:                 $inClass = false;
 674:                 $lastSeenClass = $lastSeenNamespace . ($lastSeenNamespace ? '\\' : '') . $token[1];
 675:                 $this->staticReflection[$lastSeenClass]['properties'] = array();
 676:                 $this->staticReflection[$lastSeenClass]['methods'] = array();
 677:             }
 678: 
 679:             if ($token[0] == T_NAMESPACE) {
 680:                 $lastSeenNamespace = "";
 681:                 $inNamespace = true;
 682:             } elseif ($token[0] == T_CLASS) {
 683:                 $inClass = true;
 684:             } elseif ($token[0] == T_FUNCTION) {
 685:                 if ($tokens[$i+2][0] == T_STRING) {
 686:                     $this->staticReflection[$lastSeenClass]['methods'][] = $tokens[$i+2][1];
 687:                 } elseif ($tokens[$i+2] == "&" && $tokens[$i+3][0] == T_STRING) {
 688:                     $this->staticReflection[$lastSeenClass]['methods'][] = $tokens[$i+3][1];
 689:                 }
 690:             } elseif (in_array($token[0], array(T_VAR, T_PUBLIC, T_PRIVATE, T_PROTECTED)) && $tokens[$i+2][0] != T_FUNCTION) {
 691:                 $this->staticReflection[$lastSeenClass]['properties'][] = substr($tokens[$i+2][1], 1);
 692:             }
 693:         }
 694:     }
