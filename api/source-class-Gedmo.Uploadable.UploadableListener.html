101:     /**
102:      * This event is needed in special cases where the entity needs to be updated, but it only has the
103:      * file field modified. Since we can't mark an entity as "dirty" in the "addEntityFileInfo" method,
104:      * doctrine thinks the entity has no changes, which produces that the "onFlush" event gets never called.
105:      * Here we mark the entity as dirty, so the "onFlush" event gets called, and the file is processed.
106:      *
107:      * @param \Doctrine\Common\EventArgs $args
108:      */
109:     public function preFlush(EventArgs $args)
110:     {
111:         if (empty($this->fileInfoObjects)) {
112:             // Nothing to do
113:             return;
114:         }
115: 
116:         $ea = $this->getEventAdapter($args);
117:         $om = $ea->getObjectManager();
118:         $uow = $om->getUnitOfWork();
119:         $first = reset($this->fileInfoObjects);
120:         $meta = $om->getClassMetadata(get_class($first['entity']));
121:         $config = $this->getConfiguration($om, $meta->name);
122: 
123:         foreach ($this->fileInfoObjects as $info) {
124:             $entity = $info['entity'];
125: 
126:             // If the entity is in the identity map, it means it will be updated. We need to force the
127:             // "dirty check" here by "modifying" the path. We are actually setting the same value, but
128:             // this will mark the entity as dirty, and the "onFlush" event will be fired, even if there's

137:                 }

552:             'fileSize'          => $fileInfo->getSize()

563:         } else {

565:         }
